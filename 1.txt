

SELECT audit_id,
TIMESTAMP_ADD(hstamp, INTERVAL - MOD(EXTRACT(HOUR FROM hstamp),2) HOUR) AS hstamp,
batch_id,
seq,
Array(select safe_cast(item as INT64) from unnest(split(sl.site_list_ids,':')) as item) as site_list_ids,
FROM `<var_input_table>` t
 JOIN adelphic-forecasting.temp.site_lookup_backfil sl ON  sl.site_id = t.site_id

 WHERE DATE(t.hstamp) BETWEEN '2020-12-04' AND '2020-12-15'
